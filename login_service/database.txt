// Database
CREATE DATABASE flask CHARACTER SET utf8 COLLATE utf8_unicode_ci;

// Table Users
CREATE TABLE users (id INT(6) UNSIGNED AUTO_INCREMENT PRIMARY KEY, username VARCHAR(100) NOT NULL UNIQUE, email VARCHAR(100) NOT NULL UNIQUE, password VARCHAR(100) NOT NULL, type INT(6) NOT NULL);

// Table Publishings
CREATE TABLE publishings (id INT(6) UNSIGNED AUTO_INCREMENT PRIMARY KEY, topic VARCHAR(100) NOT NULL, text TEXT(300) NOT NULL);

// Table Subscriptions
CREATE TABLE subscriptions (id INT(6) UNSIGNED AUTO_INCREMENT PRIMARY KEY, user_id INT(6) NOT NULL, topic VARCHAR(100) NOT NULL);
